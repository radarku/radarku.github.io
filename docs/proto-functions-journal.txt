title = {On the Evaluation of Space–Time Functions},

abstract ={In an environment increasingly saturated with computing devices, it is desirable for some services to be distributed, executing via local interactions between devices. Creating fast, flexible and dynamic distributed services requires a general model of function calls distributed over space–time. Prior models, however, have either depended strongly on large-scale Internet infrastructure or have restrictions in the scope or resolution of space–time for inputs, outputs or evaluation of the function. We address this by providing a formal general model of function calls over space–time. We then fully realize a practical model of space–time function calls, based in the Proto language and present both theoretical and empirical results. Finally, we show how our results for Proto generalize into implications for any model of distributed computing.},
